## Task 1: Create HUDManager.cs

**Files:**
- Create: `Assets/Scripts/MonoBehaviours/HUDManager.cs`

```csharp
using TMPro;
using Unity.Collections;
using Unity.Entities;
using UnityEngine;
using UnityEngine.UI;
using VampireSurvivors.Components;

namespace VampireSurvivors.MonoBehaviours
{
    /// <summary>
    /// Polls the ECS world each frame and updates HUD elements:
    /// per-player HP bar, XP bar, level text, plus a top-center elapsed timer.
    ///
    /// Attach to the HUD GameObject in 4_SampleScene.
    /// Wire the four panel roots and their child Image/TMP_Text refs via the Inspector arrays.
    /// </summary>
    public class HUDManager : MonoBehaviour
    {
        [Header("Player Panels (index 0-3)")]
        [SerializeField] GameObject[] panelRoots   = new GameObject[4];
        [SerializeField] Image[]      hpFillImages = new Image[4];
        [SerializeField] Image[]      xpFillImages = new Image[4];
        [SerializeField] TMP_Text[]   levelTexts   = new TMP_Text[4];
        [SerializeField] TMP_Text[]   playerLabels = new TMP_Text[4];

        [Header("Timer")]
        [SerializeField] TMP_Text timerText;

        static readonly Color HpColorHigh = new Color(0.20f, 0.80f, 0.20f, 1f);
        static readonly Color HpColorMid  = new Color(0.90f, 0.80f, 0.10f, 1f);
        static readonly Color HpColorLow  = new Color(0.85f, 0.15f, 0.15f, 1f);

        EntityQuery _playerQuery;
        bool        _queryCreated;
        float       _elapsedTime;

        void Start()
        {
            for (int i = 0; i < 4; i++)
            {
                if (playerLabels[i] != null) playerLabels[i].text = $"P{i + 1}";
                if (panelRoots[i] != null)   panelRoots[i].SetActive(false);
            }

            var world = World.DefaultGameObjectInjectionWorld;
            if (world == null) return;

            _playerQuery = world.EntityManager.CreateEntityQuery(
                ComponentType.ReadOnly<PlayerTag>(),
                ComponentType.ReadOnly<PlayerIndex>(),
                ComponentType.ReadOnly<PlayerStats>(),
                ComponentType.ReadOnly<Health>()
            );
            _queryCreated = true;
        }

        void OnDisable()
        {
            if (_queryCreated)
            {
                _playerQuery.Dispose();
                _queryCreated = false;
            }
        }

        void Update()
        {
            _elapsedTime += Time.deltaTime;
            UpdateTimer();

            var world = World.DefaultGameObjectInjectionWorld;
            if (world == null || !_queryCreated) return;

            bool[] seen = new bool[4];

            var indices = _playerQuery.ToComponentDataArray<PlayerIndex>(Allocator.Temp);
            var stats   = _playerQuery.ToComponentDataArray<PlayerStats>(Allocator.Temp);
            var healths = _playerQuery.ToComponentDataArray<Health>(Allocator.Temp);

            for (int i = 0; i < indices.Length; i++)
            {
                int slot = indices[i].Value;
                if (slot < 0 || slot >= 4) continue;
                seen[slot] = true;
                UpdatePanel(slot, in healths[i], in stats[i]);
            }

            indices.Dispose();
            stats.Dispose();
            healths.Dispose();

            for (int slot = 0; slot < 4; slot++)
                if (panelRoots[slot] != null && !seen[slot])
                    panelRoots[slot].SetActive(false);
        }

        void UpdatePanel(int slot, in Health health, in PlayerStats stats)
        {
            if (panelRoots[slot] == null) return;
            panelRoots[slot].SetActive(true);

            if (hpFillImages[slot] != null)
            {
                float hpRatio = health.Max > 0
                    ? Mathf.Clamp01((float)health.Current / health.Max)
                    : 0f;
                hpFillImages[slot].fillAmount = hpRatio;
                hpFillImages[slot].color      = HpColor(hpRatio);
            }

            if (xpFillImages[slot] != null)
            {
                float xpRatio = stats.XpToNextLevel > 0f
                    ? Mathf.Clamp01(stats.Xp / stats.XpToNextLevel)
                    : 0f;
                xpFillImages[slot].fillAmount = xpRatio;
            }

            if (levelTexts[slot] != null)
                levelTexts[slot].text = $"Lv {stats.Level}";
        }

        void UpdateTimer()
        {
            if (timerText == null) return;
            int total   = Mathf.FloorToInt(_elapsedTime);
            timerText.text = $"{total / 60:00}:{total % 60:00}";
        }

        static Color HpColor(float ratio)
        {
            if (ratio > 0.5f)  return HpColorHigh;
            if (ratio > 0.25f) return HpColorMid;
            return HpColorLow;
        }
    }
}
```

## Task 2: Build HUD hierarchy in 4_SampleScene

**Files:**
- Modify: `Assets/Scenes/4_SampleScene.unity` (via Unity Editor MCP tools)

### Canvas

Create `HUDCanvas` (UI > Canvas):
- Render Mode: Screen Space - Overlay
- Canvas Scaler: Scale With Screen Size, 1920×1080, Match 0.5
- Remove Graphic Raycaster (display-only)

### HUD root

Inside `HUDCanvas`, empty GameObject named `HUD`:
- RectTransform: stretch to fill (anchor min (0,0), anchor max (1,1), all offsets 0)
- Add `HUDManager` component

### Four Player Panels

For each N in {0,1,2,3}, create `PlayerPanel_N` inside `HUD`:

| N | Anchor Min/Max | Pivot | Pos X | Pos Y |
|---|---------------|-------|-------|-------|
| 0 | (0,0)/(0,0) | (0,0) | 20 | 20 |
| 1 | (1,0)/(1,0) | (1,0) | -20 | 20 |
| 2 | (0,1)/(0,1) | (0,1) | 20 | -20 |
| 3 | (1,1)/(1,1) | (1,1) | -20 | -20 |

Width: 220, Height: 80. Background Image color: (0,0,0,0.5).

Children of each `PlayerPanel_N`:

**`PlayerLabel`**: Anchor (0,1)→(1,1), Pivot (0,1), Left=4, Top=0, Right=-4, Height=24. TMP_Text, size 16, left-aligned, white.

**`LevelText`**: Anchor (1,1)→(1,1), Pivot (1,1), PosX=-4, PosY=0, Width=80, Height=24. TMP_Text, size 14, right-aligned, color (0.9,0.9,0.5,1). Default "Lv 1".

**`HPBarBG`**: Anchor (0,1)→(1,1), Pivot (0,1), Left=4, Top=-26, Right=-4, Height=20. Image color (0.3,0.05,0.05,0.9).
  - Child `HPFill`: anchor stretch (offsets 2px all sides). Image: Filled, Horizontal, Left origin. Color green initially.

**`XPBarBG`**: Anchor (0,1)→(1,1), Pivot (0,1), Left=4, Top=-50, Right=-4, Height=20. Image color (0.05,0.05,0.35,0.9).
  - Child `XPFill`: anchor stretch (offsets 2px all sides). Image: Filled, Horizontal, Left origin. Color (0.4,0.2,0.9,1), fillAmount=0.

### Timer

Inside `HUD`, empty GameObject named `Timer`:
- Anchor (0.5,1)→(0.5,1), Pivot (0.5,1), Width=160, Height=40, PosX=0, PosY=-10
- TMP_Text: size 24, center-aligned, white. Default "00:00".

### Wire HUDManager Inspector

On the `HUD` GameObject's `HUDManager` component:
- `Panel Roots`: [PlayerPanel_0, PlayerPanel_1, PlayerPanel_2, PlayerPanel_3]
- `Hp Fill Images`: [PlayerPanel_0/HPBarBG/HPFill, PlayerPanel_1/HPBarBG/HPFill, ...]
- `Xp Fill Images`: [PlayerPanel_0/XPBarBG/XPFill, ...]
- `Level Texts`: [PlayerPanel_0/LevelText, ...]
- `Player Labels`: [PlayerPanel_0/PlayerLabel, ...]
- `Timer Text`: Timer (TMP_Text)
